## Core Game Logic

### 1. Gravity

Gravity is applied per-column. After any shift, every column is independently "settled" — all non-null cells fall to the bottom.

```typescript
// gravity.ts

// Given a single column (array of cells, index 0=top), 
// return a new column with nulls floated to the top.
function applyGravityToColumn(col: Cell[]): Cell[] {
  const discs = col.filter(c => c !== null);
  const empties = col.filter(c => c === null);
  return [...empties, ...discs];
  // empties at top, discs at bottom — matches "gravity pulls down"
}

// Apply gravity to every column of the board.
function applyGravityAll(board: Board): Board {
  // Transpose board to work column-by-column, apply gravity, transpose back.
  const numRows = board.length;
  const numCols = board[0].length;
  
  // Extract columns
  const columns: Cell[][] = Array.from({ length: numCols }, (_, c) =>
    board.map(row => row[c])
  );
  
  const settledCols = columns.map(applyGravityToColumn);
  
  // Rebuild row-major board from settled columns
  return Array.from({ length: numRows }, (_, r) =>
    Array.from({ length: numCols }, (_, c) => settledCols[c][r])
  );
}
```

### 2. Row Shift

The key function. It operates on the flat board (absolute coordinates) and returns a new board plus reclaimed discs.

```typescript
// shift.ts

interface ShiftResult {
  board: Board;
  reclaimedDiscs: Partial<Record<Player, number>>; // discs that fell off
}

function shiftRow(
  board: Board,
  rowIndex: number,
  direction: 'left' | 'right'
): ShiftResult {
  const row = [...board[rowIndex]];
  let reclaimed: Partial<Record<Player, number>> = {};

  if (direction === 'left') {
    const ejected = row[0]; // leftmost cell falls off
    if (ejected !== null) {
      reclaimed[ejected] = (reclaimed[ejected] ?? 0) + 1;
    }
    row.shift();
    row.push(null); // vacated right end
  } else {
    const ejected = row[row.length - 1]; // rightmost cell falls off
    if (ejected !== null) {
      reclaimed[ejected] = (reclaimed[ejected] ?? 0) + 1;
    }
    row.pop();
    row.unshift(null); // vacated left end
  }

  const newBoard: Board = board.map((r, i) => (i === rowIndex ? row : [...r]));
  
  // Apply gravity to ALL columns after the shift
  const settledBoard = applyGravityAll(newBoard);
  
  return { board: settledBoard, reclaimed };
}
```

**Critical subtlety**: Gravity after a shift can cascade. For example: row 0 (top) shifts left, ejecting a disc. Now column 2 (right) has a gap at the top — no cascading needed since gravity only moves discs down, not up. However, a shift of a middle row can leave a gap in the middle of a column, causing upper discs to fall. `applyGravityAll` handles all these cases uniformly.

### 3. Validation

```typescript
// validation.ts

function canDrop(state: GameState, col: number): boolean {
  if (state.phase !== 'playing') return false;
  if (state.discs[state.currentPlayer] <= 0) return false;
  // Top row of column must be empty
  return state.board[0][col] === null;
}

function canShift(
  state: GameState,
  row: number,
  direction: 'left' | 'right'
): boolean {
  if (state.phase !== 'playing') return false;
  const currentOffset = state.rowOffsets[row];
  if (direction === 'left' && currentOffset <= -MAX_OFFSET) return false;
  if (direction === 'right' && currentOffset >= MAX_OFFSET) return false;
  // Optionally: disallow shifting an entirely empty row (no-op)
  const rowIsEmpty = state.board[row].every(cell => cell === null);
  if (rowIsEmpty) return false;
  return true;
}
```

**Note on the offset constraint**: The offset tracks how far the row has drifted from its original position, not the visual position. Shift left -> offset--. Shift right -> offset++. The constraint is `-MAX_OFFSET <= offset <= +MAX_OFFSET`. Starting at 0, a row can be shifted at most once in either direction before it must be shifted back.

```typescript
function dropDisc(state: GameState, col: number): GameState {
  // Find lowest empty row in this column
  let targetRow = -1;
  for (let r = ROWS - 1; r >= 0; r--) {
    if (state.board[r][col] === null) {
      targetRow = r;
      break;

    ...state.discs,
    [state.currentPlayer]: state.discs[state.currentPlayer] - 1,
  };

}
```

### 5. Win Detection

Win detection runs on the **settled** board after every action.

```typescript
// winDetection.ts

interface WinResult {
  winners: Player[];
  winningCells: [number, number][];
}

const LINES: [number, number][][] = [
  // Rows
  [[0,0],[0,1],[0,2]],
  [[1,0],[1,1],[1,2]],
  [[2,0],[2,1],[2,2]],
  // Columns
  [[0,0],[1,0],[2,0]],
  [[0,1],[1,1],[2,1]],
  [[0,2],[1,2],[2,2]],
  // Diagonals
  [[0,0],[1,1],[2,2]],
  [[0,2],[1,1],[2,0]],
];

function checkWin(board: Board): WinResult {
  const winners = new Set<Player>();
  const winningCells: [number, number][] = [];

  for (const line of LINES) {
    const values = line.map(([r, c]) => board[r][c]);
    if (values[0] !== null && values.every(v => v === values[0])) {
      winners.add(values[0] as Player);
      winningCells.push(...line as [number, number][]);
    }
  }

  return {
    winners: [...winners],
    winningCells: [...new Set(winningCells.map(c => JSON.stringify(c)))].map(s => JSON.parse(s)),
  };
}
```

**Both-win case**: Because we check all 8 lines on the post-action board, both players can appear in `winners`. This is handled cleanly by collecting into a Set.

### 6. Turn Advancement

```typescript
// Called at the end of every action handler
function advanceTurn(state: GameState): GameState {
  const { winners, winningCells } = checkWin(state.board);
  
  if (winners.length > 0) {
    return { ...state, phase: 'won', winners, winningCells };
  }

  // Check for draw: no valid actions left for either player
  // (all columns full, no discs remaining, and no valid shifts)
  // This is rare but theoretically possible
  const isDraw = checkDraw(state);
  if (isDraw) {
    return { ...state, phase: 'draw' };
  }

  const nextPlayer: Player = state.currentPlayer === 'red' ? 'black' : 'red';
  return { ...state, currentPlayer: nextPlayer };
}
```

### 7. The Reducer

```typescript
// gameReducer.ts

function gameReducer(state: GameState, action: Action): GameState {
  switch (action.type) {
    case 'DROP_DISC': {
      if (!canDrop(state, action.col)) return state;
      return dropDisc(state, action.col);
    }
    case 'SHIFT_ROW': {
      if (!canShift(state, action.row, action.direction)) return state;
      const { board, reclaimed } = shiftRow(state.board, action.row, action.direction);
      const newOffsets = state.rowOffsets.map((o, i) =>
        i === action.row ? o + (action.direction === 'left' ? -1 : 1) : o
      ) as RowOffsets;
      const newDiscs = { ...state.discs };
      for (const [player, count] of Object.entries(reclaimed)) {
        newDiscs[player as Player] += count as number;
      }
      return advanceTurn({ ...state, board, rowOffsets: newOffsets, discs: newDiscs });
    }
    case 'RESET_GAME':
      return initialState();
    default:
      return state;
  }
}
```

## Key Edge Cases

### 1. Gravity after shift creates a new winning line

After a row shift, gravity settles the board. This post-gravity state is what's checked for wins. For example: shifting row 1 left causes a gap in column 2 at row 1. If row 0 column 2 had a disc, it falls to row 1 col 2. If that creates three-in-a-column, the shifted player wins — even though they didn't directly place the winning disc.

### 2. Disc reclamation on shift

When a disc is shifted off the edge, it is immediately returned to its owner's count. This means a player could strategically shift off their own disc to reclaim it. The reducer must check `reclaimed` from `shiftRow` and add to `newDiscs` before calling `advanceTurn`.

### 3. Simultaneous win

After a shift, it's possible that both red and black now have three-in-a-row (e.g., a shift causes cascading gravity that completes both players' lines). The `checkWin` function collects all winning players via a Set, and the `WinOverlay` displays "Both players win!" if `winners.length === 2`.

### 4. Shifting an empty row

`canShift` returns false if a row is entirely empty. This prevents a meaningless action and avoids the offset being consumed needlessly.

### 5. Offset bookkeeping after gravity

**Important**: Row offsets track the shift of the row's cells, but gravity moves cells between rows. After gravity, a disc that "was in row 1" may now be in row 2. The offset belongs to the row slot, not the disc. So after `applyGravityAll`, offsets remain unchanged — they describe the row slot's current displacement, not the discs within it.

This means: if row 1 is at offset -1 and gravity causes a disc to fall from row 0 into row 1, that disc is now in a "shifted" row slot visually. This is correct behavior — the visual slide of row 1 still applies.

### 6. Draw condition

A draw is theoretically possible if: all columns are full, no player can shift any row (all at max/min offset), and no win has occurred. This is extremely unlikely but `checkDraw` should verify: `canDrop` is false for all columns AND `canShift` is false for all row/direction combos AND `phase !== 'won'`.

### 7. No valid actions for current player but other player can act

This is not explicitly covered by the rules, but if a player has no discs and cannot shift any row, they are stuck. A reasonable handling: auto-pass their turn, or display a "No moves available, turn passed" message. This should be detected in `advanceTurn` before switching players.

### 8. Column drop target disabled when row offsets shift columns visually

The drop targets always correspond to absolute column indices. Even though row 1 may be visually shifted right, dropping into "column 0" still means the absolute column 0. The board state is always stored in absolute coordinates, so this is consistent. Players need to understand that drop arrows refer to the grid slots, not the visual positions of shifted discs.

## Potential Gotchas

- **Immutability**: Every board operation must return new arrays. The reducer must never mutate `state.board` in place. Use `.map()` throughout.
- **Row vs. column indexing**: Be consistent. `board[row][col]`. Row 0 = top. Gravity iterates rows bottom-up.
- **Gravity direction**: Gravity always pulls toward higher row indices (row 2 = bottom). `applyGravityToColumn` collects non-null cells and places them at the end of the array (highest indices).
- **CSS clipping and the shift animation**: The board container must clip overflow so shifted rows don't show outside the grid boundary. But the shift control arrows must be outside the clip zone. Use a wrapper div for the arrows and a separate div with `overflow: hidden` for the grid.
- **React key stability**: When rendering rows with offsets, use row index as key, not content, so React doesn't re-mount rows during shifts.

```typescript
// types.ts
type Player = 'red' | 'black';
type Cell = Player | null;
type Board = Cell[][];          // board[row][col], 3×3, row 0 = top
type RowOffsets = [number, number, number];  // -1 | 0 | +1 each

interface GameState {
  board: Board;
  rowOffsets: RowOffsets;
  currentPlayer: Player;
  discs: Record<Player, number>;   // starts at 5 each

  winners: Player[];               // [] | ['red'] | ['black'] | both
```

## Core Logic

### Constants
```typescript
// constants.ts
export const ROWS = 3, COLS = 3, DISCS_PER_PLAYER = 5, MAX_OFFSET = 1;
```

### Gravity (`gravity.ts`)
Per-column: collect non-null cells, float nulls to top, discs to bottom.
`applyGravityAll` transposes the board, applies per-column, transposes back.

### Shift (`shift.ts`)
`shiftRow(board, rowIndex, 'left' | 'right')`:
- Direction left: eject `row[0]` (reclaim if non-null), shift array, push null at end.
- Direction right: eject `row[row.length-1]`, unshift null at start.
- Then call `applyGravityAll` on the whole board.
- Return `{ board, reclaimed: Partial<Record<Player, number>> }`.

### Win Detection (`winDetection.ts`)
Check all 8 lines (3 rows + 3 cols + 2 diagonals). Collect all winning players into a Set. Both can win simultaneously.

### Validation (`validation.ts`)
- `canDrop(state, col)`: phase=playing, player has discs, `board[0][col] === null`.
- `canShift(state, row, dir)`: phase=playing, offset within ±MAX_OFFSET after shift, row not entirely empty.

### Reducer (`gameReducer.ts`)
1. `DROP_DISC`: place disc at lowest empty cell, decrement disc count, `advanceTurn`.
2. `SHIFT_ROW`: call `shiftRow`, update `rowOffsets[row] ±1`, add reclaimed discs, `advanceTurn`.
3. `RESET_GAME`: return `initialState()`.

`advanceTurn`: run `checkWin`; if winners, set phase=won; else check draw; else flip `currentPlayer`.

import type { Board, Cell } from '../types';
import { ROWS, COLS } from '../constants';

function applyGravityToColumn(col: Cell[]): Cell[] {
  const discs = col.filter((c): c is NonNullable<Cell> => c !== null);
  const empties: null[] = Array(col.length - discs.length).fill(null);
  return [...empties, ...discs];
}

export function applyGravityAll(board: Board): Board {
  // Extract columns, apply gravity, rebuild row-major board
  const columns: Cell[][] = Array.from({ length: COLS }, (_, c) =>
    board.map(row => row[c])
  );
  const settled = columns.map(applyGravityToColumn);
  return Array.from({ length: ROWS }, (_, r) =>
    Array.from({ length: COLS }, (_, c) => settled[c][r])
  );
}

import type { Board, Cell, Player } from '../types';
import { applyGravityAll } from './gravity';

export interface ShiftResult {
  board: Board;
  reclaimed: Partial<Record<Player, number>>;
}

export function shiftRow(
  board: Board,
  rowIndex: number,
  direction: 'left' | 'right'
): ShiftResult {
  const row = [...board[rowIndex]];
  const reclaimed: Partial<Record<Player, number>> = {};

  let ejected: Cell;
  if (direction === 'left') {
    ejected = row[0];
    row.shift();
    row.push(null);
  } else {
    ejected = row[row.length - 1];
    row.pop();
    row.unshift(null);
  }

  if (ejected !== null) {
    reclaimed[ejected] = (reclaimed[ejected] ?? 0) + 1;
  }

  const newBoard: Board = board.map((r, i) => (i === rowIndex ? row : [...r]));
  return { board: applyGravityAll(newBoard), reclaimed };
}

import type { Board, Player } from '../types';

type Coord = [number, number];

const LINES: Coord[][] = [
  // Rows
  [[0,0],[0,1],[0,2]],
  [[1,0],[1,1],[1,2]],
  [[2,0],[2,1],[2,2]],
  // Columns
  [[0,0],[1,0],[2,0]],
  [[0,1],[1,1],[2,1]],
  [[0,2],[1,2],[2,2]],
  // Diagonals
  [[0,0],[1,1],[2,2]],
  [[0,2],[1,1],[2,0]],
];

export interface WinResult {
  winners: Player[];
  winningCells: Coord[];
}

export function checkWin(board: Board): WinResult {
  const winners = new Set<Player>();
  const winningCellSet = new Set<string>();

  for (const line of LINES) {
    const values = line.map(([r, c]) => board[r][c]);
    if (values[0] !== null && values.every(v => v === values[0])) {
      winners.add(values[0] as Player);
      for (const coord of line) {
        winningCellSet.add(JSON.stringify(coord));
      }
    }
  }

  return {
    winners: [...winners],
    winningCells: [...winningCellSet].map(s => JSON.parse(s) as Coord),
  };
}

import type { GameState, Action, Player, Board, RowOffsets } from '../types';
import { ROWS, COLS, DISCS_PER_PLAYER } from '../constants';
import { applyGravityAll } from './gravity';
import { shiftRow } from './shift';
import { checkWin } from './winDetection';
import { canDrop, canShift } from './validation';

function emptyBoard(): Board {
  return Array.from({ length: ROWS }, () => Array(COLS).fill(null));
}

export function initialState(): GameState {
  return {
    board: emptyBoard(),
    rowOffsets: [0, 0, 0],
    currentPlayer: 'red',
    discs: { red: DISCS_PER_PLAYER, black: DISCS_PER_PLAYER },
    phase: 'playing',
    winners: [],
    winningCells: [],
  };
}

function otherPlayer(p: Player): Player {
  return p === 'red' ? 'black' : 'red';
}

function hasMoves(state: GameState, player: Player): boolean {
  const s = { ...state, currentPlayer: player };
  for (let c = 0; c < COLS; c++) {
    if (canDrop(s, c)) return true;
  }
  for (let r = 0; r < ROWS; r++) {
    if (canShift(s, r, 'left') || canShift(s, r, 'right')) return true;
  }
  return false;
}

function advanceTurn(state: GameState): GameState {
  const { winners, winningCells } = checkWin(state.board);
  if (winners.length > 0) {
    return { ...state, phase: 'won', winners, winningCells };
  }

  const next = otherPlayer(state.currentPlayer);

  // Check if next player has any moves; if not, check if current can continue
  if (!hasMoves({ ...state, currentPlayer: next }, next)) {
    if (!hasMoves(state, state.currentPlayer)) {
      return { ...state, phase: 'draw' };
    }
    // Next player is stuck — skip their turn (stay with current player)
    return state;
  }

  return { ...state, currentPlayer: next };
}

function dropDisc(state: GameState, col: number): GameState {
  let targetRow = -1;
  for (let r = ROWS - 1; r >= 0; r--) {
    if (state.board[r][col] === null) {
      targetRow = r;
      break;
    }
  }
  if (targetRow === -1) return state;

  const newBoard = state.board.map((row, r) =>
    row.map((cell, c) => (r === targetRow && c === col ? state.currentPlayer : cell))
  );

  // Apply gravity after drop (disc already falls to lowest empty, but just in case)
  const settledBoard = applyGravityAll(newBoard);

  const newDiscs = {
    ...state.discs,
    [state.currentPlayer]: state.discs[state.currentPlayer] - 1,
  };

  return advanceTurn({ ...state, board: settledBoard, discs: newDiscs });
}

export function gameReducer(state: GameState, action: Action): GameState {
  switch (action.type) {
    case 'DROP_DISC': {
      if (!canDrop(state, action.col)) return state;
      return dropDisc(state, action.col);
    }
    case 'SHIFT_ROW': {
      if (!canShift(state, action.row, action.direction)) return state;
      const { board, reclaimed } = shiftRow(state.board, action.row, action.direction);
      const newOffsets = state.rowOffsets.map((o, i) =>
        i === action.row ? o + (action.direction === 'left' ? -1 : 1) : o
      ) as RowOffsets;
      const newDiscs = { ...state.discs };
      for (const [player, count] of Object.entries(reclaimed) as [Player, number][]) {
        newDiscs[player] += count;
      }
      return advanceTurn({ ...state, board, rowOffsets: newOffsets, discs: newDiscs });
    }
    case 'RESET_GAME':
      return initialState();
  }
}

import { useReducer, useState, useRef, useEffect } from 'react';
import type { Board as BoardType, RowOffsets } from './types';
import { gameReducer, initialState } from './logic/gameReducer';
import { Board } from './components/Board';
import { DiscCounter } from './components/DiscCounter';
import { PlayerBanner } from './components/PlayerBanner';
import './App.css';

export function App() {
  const [gameState, dispatch] = useReducer(gameReducer, undefined, initialState);

  // Display state — what the board actually renders. During a shift animation,
  // displayBoard holds the PRE-shift board while displayOffsets reflects the
  // new (post-shift) offset so the CSS transition plays. After the transition
  // ends, displayBoard snaps to gameState.board.
  const [displayBoard, setDisplayBoard] = useState<BoardType>(() => initialState().board);
  const [displayOffsets, setDisplayOffsets] = useState<RowOffsets>([0, 0, 0]);

  const shiftAnimatingRef = useRef(false);
  const animatingRowRef = useRef<number | null>(null);
  // Always-fresh refs so callbacks aren't stale
  const latestBoardRef = useRef(gameState.board);
  const latestOffsetsRef = useRef(gameState.rowOffsets);
  latestBoardRef.current = gameState.board;
  latestOffsetsRef.current = gameState.rowOffsets;

  // Sync display state for drops and resets (any action that isn't a shift)
  useEffect(() => {
    if (!shiftAnimatingRef.current) {
      setDisplayBoard(gameState.board);
      setDisplayOffsets(gameState.rowOffsets);
    }
  }, [gameState.board, gameState.rowOffsets]);

  function handleDrop(col: number) {
    dispatch({ type: 'DROP_DISC', col });
    // useEffect above will sync displayBoard after re-render
  }

  function handleShift(row: number, direction: 'left' | 'right') {
    const preBoard = gameState.board; // capture before dispatch
    dispatch({ type: 'SHIFT_ROW', row, direction });

    shiftAnimatingRef.current = true;
    animatingRowRef.current = row;

    // Keep showing the pre-shift board; only change offsets to trigger the CSS transition
    setDisplayBoard(preBoard);
    setDisplayOffsets(prev => {
      const next = [...prev] as RowOffsets;
      next[row] = prev[row] + (direction === 'left' ? -1 : 1);
      return next;
    });
  }

  function handleShiftTransitionEnd(row: number) {
    if (row !== animatingRowRef.current || !shiftAnimatingRef.current) return;
    shiftAnimatingRef.current = false;
    animatingRowRef.current = null;
    // Snap board to settled post-shift state; any gravity-dropped discs appear here
    setDisplayBoard(latestBoardRef.current);
    setDisplayOffsets(latestOffsetsRef.current);
  }

  function handleReset() {
    dispatch({ type: 'RESET_GAME' });
    shiftAnimatingRef.current = false;
    animatingRowRef.current = null;
    // useEffect will sync on next render
  }

  return (
    <div className="game">
      <h1 className="title">Shift Tac Toe</h1>

      <div className="counters">
        <DiscCounter
          player="red"
          count={gameState.discs.red}
          isCurrentPlayer={gameState.currentPlayer === 'red'}
        />
        <DiscCounter
          player="black"
          count={gameState.discs.black}
          isCurrentPlayer={gameState.currentPlayer === 'black'}
        />
      </div>

      <PlayerBanner
        currentPlayer={gameState.currentPlayer}
        phase={gameState.phase}
        winners={gameState.winners}
      />

      <Board
        gameState={gameState}
        displayBoard={displayBoard}
        displayOffsets={displayOffsets}
        onDrop={handleDrop}
        onShift={handleShift}
        onShiftTransitionEnd={handleShiftTransitionEnd}
      />

      <button className="new-game-btn" onClick={handleReset}>
        New Game
      </button>
    </div>
  );
}

import { useRef, useLayoutEffect, useMemo } from 'react';
import type { Cell as CellType, GameState, Board as BoardType, RowOffsets } from '../types';
import { ROWS, COLS, CELL_WITH_GAP } from '../constants';
import { canDrop, canShift } from '../logic/validation';
import { Cell } from './Cell';
import { ColumnDropTarget } from './ColumnDropTarget';
import { RowShiftControls } from './RowShiftControls';

interface BoardProps {
  gameState: GameState;
  displayBoard: BoardType;
  displayOffsets: RowOffsets;
  onDrop: (col: number) => void;
  onShift: (row: number, direction: 'left' | 'right') => void;
  onShiftTransitionEnd: (row: number) => void;
}

// Build the 5-slot array for a row based on its current offset.
// offset  0 → board data at slots [1,2,3],  transform shows slots 1-3
// offset -1 → board data at slots [2,3,4],  transform shows slots 2-4
// offset +1 → board data at slots [0,1,2],  transform shows slots 0-2
function getSlots(row: CellType[], offset: number): CellType[] {
  const slots: CellType[] = [null, null, null, null, null];
  const start = 1 - offset; // which slot index board[0] maps to
  for (let i = 0; i < COLS; i++) {
    slots[start + i] = row[i];
  }
  return slots;
}

export function Board({
  gameState,
  displayBoard,
  displayOffsets,
  onDrop,
  onShift,
  onShiftTransitionEnd,
}: BoardProps) {
  const { winningCells } = gameState;

  // Track previous displayBoard to detect newly placed discs for drop animation
  const prevBoardRef = useRef<BoardType>(displayBoard);
  const newCells = useMemo(() => {
    const result = new Set<string>();
    for (let r = 0; r < ROWS; r++) {
      for (let c = 0; c < COLS; c++) {
        if (!prevBoardRef.current[r][c] && displayBoard[r][c]) {
          result.add(`${r}-${c}`);
        }
      }
    }
    return result;
  }, [displayBoard]);

  useLayoutEffect(() => {
    prevBoardRef.current = displayBoard;
  }, [displayBoard]);

  function isWinning(r: number, c: number) {
    return winningCells.some(([wr, wc]) => wr === r && wc === c);
  }

  return (
    <div className="board-area">
      {/* Drop target arrows — aligned to the 3-wide frame */}
      <div className="drop-targets">
        <div className="shift-spacer" />
        {Array.from({ length: COLS }, (_, c) => (
          <ColumnDropTarget
            key={c}
            col={c}
            disabled={!canDrop(gameState, c)}
            onDrop={() => onDrop(c)}
          />
        ))}
        <div className="shift-spacer" />
      </div>

      {/* Rows */}
      <div className="board-rows">
        {Array.from({ length: ROWS }, (_, r) => {
          const offset = displayOffsets[r];
          const slots = getSlots(displayBoard[r], offset);
          // transform: show the correct 3 slots through the frame
          const transform = `translateX(${-(1 - offset) * CELL_WITH_GAP}px)`;

          return (
            <div key={r} className="board-row-wrapper">
              <RowShiftControls
                row={r}
                side="left"
                canShiftLeft={canShift(gameState, r, 'left')}
                canShiftRight={canShift(gameState, r, 'right')}
                onShiftLeft={() => onShift(r, 'left')}
                onShiftRight={() => onShift(r, 'right')}
              />

              {/* The frame clips to exactly 3-cells wide; the slider has 5 cells */}
              <div className="row-frame">
                <div
                  className="row-slider"
                  style={{ transform }}
                  onTransitionEnd={(e) => {
                    if (e.propertyName === 'transform') onShiftTransitionEnd(r);
                  }}
                >
                  {slots.map((cell, slotIdx) => {
                    // Map slot index back to absolute board column for win/new checks
                    const boardCol = slotIdx - (1 - offset);
                    const inBounds = boardCol >= 0 && boardCol < COLS;
                    const winning = inBounds && isWinning(r, boardCol);
                    const isNew = inBounds && newCells.has(`${r}-${boardCol}`);
                    return (
                      <Cell
                        key={isNew ? `${r}-${slotIdx}-${cell}-new` : `${r}-${slotIdx}`}
                        player={cell}
                        isWinning={winning}
                        animate={isNew}
                      />
                    );
                  })}
                </div>
              </div>

              <RowShiftControls
                row={r}
                side="right"
                canShiftLeft={canShift(gameState, r, 'left')}
                canShiftRight={canShift(gameState, r, 'right')}
                onShiftLeft={() => onShift(r, 'left')}
                onShiftRight={() => onShift(r, 'right')}
              />
            </div>
          );
        })}
      </div>
    </div>
  );
}

  // displayBoard: what cells are rendered. Frozen at pre-shift state during animation.

      setTransformOffsets(gameState.rowOffsets);

    }
